    call、bind和apply中this指向和手写方法

    Amclic 2021/4/16 JavaScript面经

    # 什么是this？

    this 就是一个对象。不同情况下 this 指向的不同，有以下几种情况，（亲自测试一下，这样会更容易弄懂）

    # this的指向问题

    不同情况下的this指向（4 种）

    # 1. 对象调用

    this 指向该对象（前边谁调用 this 就指向谁）。

    var obj = {
	name:'张三',
	age: '18',
	print: function(){
		console.log(this)
		console.log(this.name + ':' + this.age)
	}
}

// 通过对象的方式调用函数
obj.print(); 	// this 指向 obj

    this 永远指向新创建的对象。

	this.age = age;
	console.log(this);

var test = new Person('张三',22); // this = > test

    # 4. 箭头函数中的 this

    由于箭头函数没有单独的this 值。箭头函数的this 与声明所在的上下文相同。也就是说调用箭头函数的时候，不会隐式的调用this 参数，而是从定义时的函数继承上下文。

    const obj = {
	a:()=>{
		console.log(this);
	}
}

// 对象调用箭头函数
obj.a(); 	// window

    # 如何改变this的指向？

    我们可以通过调用函数的 call、apply、bind 来改变 this 的指向。

    • 这三个方法有什么共同点和不同点？
    var obj = {
	name:'张三',
	age:'18',
	nickName:'法外狂徒'
}

function print(){
	console.log(this); // 打印 this 的指向
	console.log(arguments); // 打印传递的参数
}

// 通过 call 改变 this 指向
print.call(obj,1,2,3);

// 通过 apply 改变 this 指向
print.apply(obj,[1,2,3]);

// 通过 bind 改变 this 的指向
let fn = print.bind(obj,1,2,3);
fn()

    共同点和不同点

    共同点 不同点
    call 能改变this指向 传参是单个传递，函数直接执行
    bind 第一个传递参数为this的指向对象 数组传递参数，函数直接执行
    apply 都采用后续传参的形式 可以数组/值传参，函数需要调用

    # call、apply、bind的实现与手写方法

    说说 call、apply、bind的内部实现？

    # 1. 手写call

    • 首先 context 为可选参数，如果不传的话默认上下文为 window ；
    • 接下来给 context 创建一个 fn 属性，并将值设置为需要调用的函数；
    • 因为 call 可以传入多个参数作为调用函数的参数，所以需要将参数剥离出来；
    • 然后调用函数并将对象上的函数删除。
    // this 为调用的函数
// context 是参数对象
Function.prototype.myCall = function(context){
	// 判断调用者是否为函数
	if(typeof this !== 'function'){
		throw new TypeError('Error')
	}
	// 不传参默认为 window
	context = context || window
	// 新增 fn 属性,将值设置为需要调用的函数
	context.fn = this
	// 将 arguments 转化为数组将 call 的传参提取出来 [...arguments]
	const args = Array.from(arguments).slice(1)
	// 传参调用函数
	const result = context.fn(...args)
	// 删除函数
	delete context.fn
	// 返回执行结果
	return result;
}
// 普通函数
function print(age){
	console.log(this.name+" "+age);
}
// 自定义对象
var obj = {
	name:'张三'
}
// 调用函数的 call 方法
print.myCall(obj,1,2,3)

    # 2. 手写apply

    • 首先 context 为可选参数，如果不传的话默认上下文为 window
    • 接下来给 context 创建一个 fn 属性，并将值设置为需要调用的函数
    • 因为 apply 传参是数组传参，所以取得数组，将其剥离为顺序参数进行函数调用
    • 然后调用函数并将对象上的函数删除
    // 手写一个 apply 方法
Function.prototype.myApply = function(context){
	// 判断调用者是否为函数
	if(typeof this !== 'function'){
		throw new TypeError('Error')
    }
	// 不传参默认为 window
	context = context || window
	// 新增 fn 属性,将值设置为需要调用的函数
	context.fn = this
	// 返回执行结果
	let result;
	// 判断是否有参数传入
	if(arguments[1]){
		result = context.fn(...arguments[1])
	}else{
		result = context.fn()
	}
	// 删除函数
	delete context.fn
	// 返回执行结果
	return result;
}
// 普通函数
function print(age,age2,age3){
	console.log(this.name+" "+ age + " "+ age2+" "+age3);
}
// 自定义对象
var obj = {
	name:'张三'
}
// 调用函数的 call 方法
print.myApply(obj,[1,2,3])

    # 3. 手写bind

    • 判断调用者是否为函数。
    • 截取参数，注意：这里有两种形式传参。
    • 返回一个函数，判断外部哪种方式调用了该函数（new | 直接调用）
    // 手写一个 bind 函数
Function.prototype.myBind = function (context) {
// 判断调用者是否为函数
	if(typeof this !== 'function'){
		throw new TypeError('Error')
	}
	// 截取传递的参数
	const args = Array.from(arguments).slice(1)
	// _this 指向调用的函数
	const _this = this;
	// 返回一个函数
	return function F(){
		// 因为返回了一个函数，我们可以 new F()，所以需要判断
		// 对于 new 的情况来说，不会被任何方式改变 this
		if(this instanceof F){
			return new _this(...args,...arguments)
		}else{
		return _this.apply(context,args.concat(...arguments))
		}
	}
}

// 普通函数
function print(){
    // new 的方式调用 bind 参数输出换做 [...arguments]
    console.log(this.name);
}
// 自定义对象
var obj = {
	name:'张三'
}
// 调用函数的 call 方法
let F = print.myBind(obj,1,2,3);
// 返回对象
let obj1 = new F();
console.log(obj1);
